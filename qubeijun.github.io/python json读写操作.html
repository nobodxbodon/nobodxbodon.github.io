<!--more--> <p>python json 读写操作</p><p>1、dumps把字典类型转换成字符串</p>

{% code %} import json

new_dict = {'s_uid': 1, 's_date': ['2017-05-01', '2017-05-15'], 's_T': 2, 's_time': ['12:00:00', '17:00:00'], 's_local': 2, 'percent': 0.53, 'loss': 0.3, 'a': 4}

print(new_dict) print(type(new_dict))

json_str = json.dumps(new_dict) print(json_str) print(type(json_str)) {% endcode %}

<p>结果： </p>

{% code %} {'a': 4, 'loss': 0.3, 's_date': ['2017-05-01', '2017-05-15'], 's_uid': 1, 's_time': ['12:00:00', '17:00:00'], 's_local': 2, 'percent': 0.53, 's_T': 2}

<type 'dict'> {"a": 4, "loss": 0.3, "s_date": ["2017-05-01", "2017-05-15"], "s_uid": 1, "s_time": ["12:00:00", "17:00:00"], "s_local": 2, "percent": 0.53, "s_T": 2}

<type 'str'> {% endcode %}
