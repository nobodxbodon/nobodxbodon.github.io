网络安全

# 显示网络连接\路由表和网络接口信息
netstat - 
    netstat [选项]
        # 显示所有socket,包括正在的监听的
        -a
        # 每个一秒就重新显示一遍,直到用户中断它
        -c
        # 显示每个协议统计信息
        -s
        # 显示以太网统计信息,此选项可以与 -s 结合
        -e
        # 显示当前连接卸载状态
        -t
        # 显示 NetworkDirect 连接、侦听器和共享
        -x
        # 显示所有连接的TCP连接模板
        -y
        # 显示正在进行的工作
        -v

# 查看80端口占用情况
netstat -aon|findstr '80'
    TCP  0.0.0.0:80  0.0.0.0:0  LISTENING  6436
# 查找PID对应的进程
tasklist|findstr '6436'
    httpd.exe  6436 Services  0  12,040 K
# 结束名为 httpd.exe 的进程
taskkill /f /t im httpd.exe
