下面是中文化后的src/maple_driver/src/maple.cpp代码:

using namespace maple;
void 报错(int 返回) {
  switch (返回) {
    case 无误:
    case 退出帮助:
      break;
    case 退出:
      错误(错误日志, "因错误退出!");
      break;
    case 非法参数:
      错误(错误日志, "非法参数!");
      break;
    case 初始化失败:
      错误(错误日志, "初始化失败!");
      break;
    case 文件未找到:
      错误(错误日志, "文件未找到!");
      break;
    case 工具未找到:
      错误(错误日志, "工具未找到!");
      break;
    case 编译失败:
      错误(错误日志, "编译失败!");
      break;
    case 未实现:
      错误(错误日志, "未实现!");
      break;
    default:
      break;
  }
}

using namespace maple;

int main(int argc, char **argv) {
  选项类 选项;
  int 返回 = 选项.分析(argc, argv);
  if (返回 == 错误码::无误) {
    返回 = 编译器工厂类::取个例().编译(选项);
  }
  报错(返回);
  return 返回;
}
