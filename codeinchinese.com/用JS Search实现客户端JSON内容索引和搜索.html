var 重建索引 = function() {
  搜索 = new JsSearch.Search('包');

  搜索.tokenizer = new JsSearch.StopWordsTokenizer(搜索.tokenizer);
  搜索.indexStrategy = new JsSearch.AllSubstringsIndexStrategy();
  
  搜索.addIndex('类');
  搜索.addIndex('名');

  搜索.addDocuments(所有数据);
};
